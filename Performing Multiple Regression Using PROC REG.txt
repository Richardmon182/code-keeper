/*Performing Multiple Regression Using PROC REG*/
/*Run a regression of PctBodyFat2 on the corresponding variables*/
/*st103s02.sas*/  /*Part A*/
ods graphics off;
proc reg data=STAT1.BodyFat2;
    model PctBodyFat2=Age Weight Height
          Neck Chest Abdomen Hip Thigh
          Knee Ankle Biceps Forearm Wrist;
    title 'Regression of PctBodyFat2 on All '
          'Predictors';
run;
quit;

/* Eliminate the variable with the highest p-value*/
/* Knee was removed because it has the largest p-value (0.9552)*/
/*st103s02.sas*/  /*Part B*/
ods graphics off;
proc reg data=STAT1.BodyFat2;
    model PctBodyFat2=Age Weight Height
          Neck Chest Abdomen Hip Thigh
          Ankle Biceps Forearm Wrist;
    title 'Regression of PctBodyFat2 on All '
          'Predictors, Minus Knee';
run;
quit;


/* Chest was removed because it is the variable with the highest p-value*/
/*st103s02.sas*/  /*Part C*/
ods graphics off;
proc reg data=STAT1.BodyFat2;
    model PctBodyFat2=Age Weight Height
          Neck Abdomen Hip Thigh
          Ankle Biceps Forearm Wrist;
    title 'Regression of PctBodyFat2 on All '
          'Predictors, Minus Knee, Chest';
run;
quit;
